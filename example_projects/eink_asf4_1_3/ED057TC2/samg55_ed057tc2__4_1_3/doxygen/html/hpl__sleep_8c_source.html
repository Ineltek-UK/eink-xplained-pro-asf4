95  }
96  }

105  }

134 {
135  uint32_t mor;
136  uint32_t mckr;
137  uint32_t pll_sr = 0;
138 
139  /* Switch MAINCK to external XTAL */
140  if (CKGR_MOR_MOSCXTBY == (_mor & CKGR_MOR_MOSCXTBY)) {
141  /* Bypass mode */
142  /* Select bypass mode */
143  mor = hri_pmc_read_CKGR_MOR_reg(PMC);
144  mor = (mor & (~CKGR_MOR_MOSCXTEN)) | CKGR_MOR_KEY_PASSWD | CKGR_MOR_MOSCXTBY | CKGR_MOR_MOSCSEL;
145  hri_pmc_write_CKGR_MOR_reg(PMC, mor);
146  /* Fully restore (Disable XTALs) */
147  hri_pmc_write_CKGR_MOR_reg(PMC, _mor | CKGR_MOR_KEY_PASSWD);
148  } else if (CKGR_MOR_MOSCXTEN == (_mor & CKGR_MOR_MOSCXTEN)) {
149  /* Enable External XTAL */

152  mor = (mor & (~CKGR_MOR_MOSCXTBY)) | CKGR_MOR_MOSCXTEN;
153  hri_pmc_write_CKGR_MOR_reg(PMC, mor);
154  while (!hri_pmc_get_SR_reg(PMC, PMC_SR_MOSCXTS)) {
155  /* Wait the XTAL to stabilize */
156  }
157  }
158 
159  /* Select External XTAL */

215 void _go_to_sleep(void)
216 {
217  bool irq_enabled = (__get_PRIMASK() == 0);
218 
219  if (_sleep_mode == 1 || _sleep_mode == 2) {
220  __disable_irq();
221  _save_clocks();
222  __enable_irq();
223 
224  /* Flash in wait mode: PMC_FSMR_FLPM_FLASH_STANDBY/_DEEP_POWERDOWN */
225  hri_pmc_write_FSMR_FLPM_bf(PMC, _sleep_mode - 1);
226  /* Set the WAITMODE bit = 1 */
227  __DSB();
228  hri_pmc_set_CKGR_MOR_reg(PMC, CKGR_MOR_KEY_PASSWD | CKGR_MOR_WAITMODE);
229  while (!hri_pmc_get_SR_reg(PMC, PMC_SR_MCKRDY)) {
230  /* Waiting for Master Clock Ready MCKRDY = 1 */
231  }
232  /* Waiting for MOSCRCEN bit cleared is strongly recommended
233  * to ensure that the core will not execute undesired instructions
234  */
235  while (!hri_pmc_get_CKGR_MOR_reg(PMC, CKGR_MOR_MOSCRCEN)) {
236  /* Waiting for MOSCRCEN */
237  }
238  /* Flash in idle mode: PMC_FSMR_FLPM_FLASH_IDLE */
239  hri_pmc_write_FSMR_FLPM_bf(PMC, 2);
240 
241  __disable_irq();
242  _restore_clocks();
243 
244  if (irq_enabled) {
245  __enable_irq();
246  }
247  } else {
248  __DSB();
249  __WFI();
250  }
251 }
