190 {
191  eink_coordinate x1_set, y1_set, odd_even_bit;
192  uint32_t byte_set;
193 
194  /* Ignore any pixels being set outside of the display window */
195  if ( (x_set >= 0) && (x_set < uc8177_global_instance.display_width) && (y_set >= 0) && (y_set < uc8177_global_instance.display_height) ) {
196 
197  if ( (uc8177_global_instance.display_config.display_rotation == ROTATE_0) || (uc8177_global_instance.display_config.display_rotation == ROTATE_180) ) {
198  /* Calculate which byte the pixel in question is contained in */
199  y1_set = (y_set - (y_set % 2)) / 2;
200  byte_set = y1_set + ((GFX_ED028TC1_MAX_WIDTH / 2) * x_set);
201  odd_even_bit = y_set % 2;
202  } else {
203  /* Calculate which byte the pixel in question is contained in */
204  x1_set = (x_set - (x_set % 2)) / 2;
205  byte_set = x1_set + ((GFX_ED028TC1_MAX_WIDTH / 2) * y_set);
206  odd_even_bit = x_set % 2;
207  }
208  if (odd_even_bit) {
209  //Set bits 3:0
210  eink_set_bit(&ptr_eink_gfx_config->display_buffer_1_ptr[byte_set], 3, ((pixel_set >> 3) & 0x1));
211  eink_set_bit(&ptr_eink_gfx_config->display_buffer_1_ptr[byte_set], 2, ((pixel_set >> 2) & 0x1));
212  eink_set_bit(&ptr_eink_gfx_config->display_buffer_1_ptr[byte_set], 1, ((pixel_set >> 1) & 0x1));
213  eink_set_bit(&ptr_eink_gfx_config->display_buffer_1_ptr[byte_set], 0, (pixel_set & 0x1));
214  } else {
215  //Set bits 6:4
