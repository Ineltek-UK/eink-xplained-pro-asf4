195  uint8_t bit_set, pixel_set_1 = 1, pixel_set_2 = 1;
196  uint8_t height_bytes = uc8151_global_instance.display_height / 8;
197 
198  /* Ignore any pixels being set outside of the display window */
199  if ( (x_set >= 0) && (x_set < uc8151_global_instance.display_width) && (y_set >= 0) && (y_set < uc8151_global_instance.display_height) ) {
200 
201  if ( (uc8151_global_instance.panel_settings.display_rotation == ROTATE_90) || (uc8151_global_instance.panel_settings.display_rotation == ROTATE_270) ) {
202  /* Calculate which byte the pixel in question is contained in */
203  x1_set = 15 - ((x_set - (x_set % 8)) / 8);
204  byte_set = (((y_set + 1) * height_bytes) - 1 - x1_set);
205  /* Calculate which bit in that byte the pixel in question is */
206  bit_set = 7 - (x_set % 8);
207  } else {
208  /* Calculate which byte the pixel in question is contained in */
209  y1_set = ((y_set - (y_set % 8)) / 8);
210  byte_set = (((x_set + 1) * height_bytes) - 1 - y1_set);
211  /* Calculate which bit in that byte the pixel in question is */
212  bit_set = (y_set % 8);
213  }
230  if (uc8151_global_instance.panel_settings.display_colours == B_W_PIXELS) {
231 
232  /*
233  * DTM2
234  * BLACK: 0 (BLACK BORDER)
235  * WHITE: 1 (BLACK BORDER)
236  */
237  if (pixel_colour == PIXEL_BLACK) pixel_set_2 = 0;
238  else if (pixel_colour == PIXEL_WHITE) pixel_set_2 = 1;
239  /* Ignore RED pixels if not supported */
240 
241  /* Set the NEW display buffer */
242  eink_set_bit(&ptr_eink_gfx_config->display_buffer_2_ptr[byte_set], bit_set, pixel_set_2);
243 
244  } else {
245 
246  /*
247  * DTM1 DTM2
248  * BLACK: 0 1 (WHITE BORDER)
249  * WHITE: 1 1 (WHITE BORDER)
250  * RED: 1 0 (WHITE BORDER)
251  * RED: 0 0 (WHITE BORDER)
252  */
253  if (pixel_colour == PIXEL_BLACK) {
254  pixel_set_1 = 0;
255  pixel_set_2 = 1;
256  } else if(pixel_colour == PIXEL_WHITE) {
257  pixel_set_1 = 1;
258  pixel_set_2 = 1;
259  } else if((pixel_colour == PIXEL_RED) || (pixel_colour == PIXEL_YELLOW)) {
260  pixel_set_1 = 0;
261  pixel_set_2 = 0;
262  }
263 
264  /* Set the B/W and RED display buffers */
265  eink_set_bit(&ptr_eink_gfx_config->display_buffer_1_ptr[byte_set], bit_set, pixel_set_1);
